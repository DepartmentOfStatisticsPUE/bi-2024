N <- 38e6
f_share <- 0.8
bias <- function(x, n_size=N*f_share, D0=(1-f_share)/f_share) sqrt(n_size)*sqrt(D0*x^2 /(1-D0*x^2))
curve(
  expr = bias,
  from = 0.0001,
  to = 0.006,
  ylab = "Błąd standardowy",
  xlab = expression(rho),
  ylim = c(0, 15),
  main = "Ilustracja big data paradox (Meng 2018, N = 38 mln)"
)
